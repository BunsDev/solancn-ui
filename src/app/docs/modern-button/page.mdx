# Section: Copy Button
 
 This section demonstrates a copy button component with two tabs:
 Preview: Displays the live interactive copy button component.
 Code: Shows the source code for the copy button component.



<Tabs defaultValue="preview" className="">
  <TabsList>
    <TabsTrigger value="preview">Preview</TabsTrigger>
    <TabsTrigger value="code">Code</TabsTrigger>
  </TabsList>

  {/* Preview tab renders the interactive CopyButtonView component */}
  <TabsContent value="preview">
    <ComponentRenderer component={<CopyButtonView />} />
  </TabsContent>

  {/* Code tab displays the source code of the CopyButtonView component */}
  <TabsContent value="code">
    <CodeBlock filePath="src/app/docs/modern-button/button.tsx" />
  </TabsContent>
</Tabs>
