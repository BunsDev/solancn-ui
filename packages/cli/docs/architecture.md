
```
solancn
├── init          # Initialize a new project with SolancnUI
├── components    # Manage components
│   ├── list      # List available components
│   ├── add       # Add component to project
│   └── remove    # Remove component from project
├── blocks        # Manage blocks
│   ├── list      # List available blocks
│   ├── add       # Add block to project
│   └── preview   # Preview block in browser
├── theme         # Manage theme
│   ├── customize # Customize theme tokens
│   └── export    # Export theme configuration
└── dev           # Start development server
```
